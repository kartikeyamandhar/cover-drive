# GPU stack, installed on the RunPod pod ONLY (needs CUDA). The Mac never installs these.
# IMPORTANT: a plain `unsloth` install pulls the LATEST torch (a 2.10 preview that wants
# 2.11), which mismatches the RunPod cu124 image's stable torch 2.4.1. That mismatch makes
# Unsloth skip its cpp extensions and fall back to a Python path whose tokenizer can't be
# pickled by datasets ('cannot pickle ConfigModuleInstance'). The torch-matched Unsloth
# build keeps torch at the image's 2.4.1+cu124 and the whole stack coherent. train.py is
# robust to the TRL version unsloth pulls (it auto-detects the kwarg names).
unsloth[cu124-torch240]
peft
bitsandbytes
accelerate
datasets
wandb
# Repo runtime deps that unsloth does not provide:
pydantic-settings>=2.6,<3
structlog>=24.4,<25
huggingface-hub>=0.25,<1
